Conditional Compilation

  • main.go
  • main.386.go - // +build 386
  • main.amd64.go - // +build amd64
  • main.darwin.go - // +build darwin
  • main.nix.go - // +build !windows
  • main.windows.go - // +build windows

func getMagicFilePath() string {

main.nix.go


// +build !windows

const fileExtension string = ""

func getUserHomeDir() string {
  return os.Getenv("HOME")
}

main.windows.go


// +build windows

const fileExtension string = ".exe"

func getUserHomeDir() string {
  return os.Getenv("USERPROFILE")
}
